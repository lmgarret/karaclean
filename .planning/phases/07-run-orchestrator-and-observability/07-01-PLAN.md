    - dry-run: dryRun=true passed through to ExecuteAction, RunSummary counts archive/delete normally
    - mixed scenario: 3 bookmarks (1 archived, 1 excepted, 1 no-match) -> correct split counts

    - output contains "archived=", "deleted=", "no_match=", "excepted=", "errors=" with correct values
  </behavior>
  <implementation>
    RunSummary struct: exported int fields (Archived, Deleted, NoMatch, Excepted, Errors) with json tags.
    RunSummary.String() method: key=value format for log.Printf.
    Run(ctx, api, rules, dryRun) (RunSummary, error):
      1. runTime := time.Now()
      2. bookmarks, err := api.ListBookmarks(ctx) -- fail-fast on error
      3. log.Printf("collected %d bookmarks", len(bookmarks))
      4. For each bookmark, for each rule: MatchesConditions -> if no, continue to next rule.
         If yes, check MatchesExceptions -> if excepted, summary.Excepted++, matched=true, break.
         If not excepted, ExecuteAction -> if result.Err != nil, summary.Errors++ else increment Archived/Deleted.
         matched=true, break (first-match-wins).
      5. If !matched after all rules, summary.NoMatch++
      6. Return summary, nil

```go
func Run(ctx context.Context, api KarakeepAPI, rules []config.Rule, dryRun bool) (RunSummary, error) {
    runTime := time.Now()

    // Phase 1: Collect all bookmarks (fail-fast on error)
    bookmarks, err := api.ListBookmarks(ctx)
    if err != nil {
        return RunSummary{}, fmt.Errorf("collecting bookmarks: %w", err)
    }
    log.Printf("collected %d bookmarks", len(bookmarks))

    // Phase 2: Evaluate rules and act
    var summary RunSummary
    for _, b := range bookmarks {
        matched := false
        for _, rule := range rules {
            if !MatchesConditions(b, rule.Conditions, runTime) {
                continue
            }
            if MatchesExceptions(b, rule.Unless) {
                summary.Excepted++
                matched = true
                break
            }
            result := ExecuteAction(ctx, api, rule.Action, b.ID, rule.Name, dryRun)
            if result.Err != nil {
                summary.Errors++
            } else {
                switch rule.Action {
                case "archive":
                    summary.Archived++
                case "delete":
                    summary.Deleted++
                }
            }
            matched = true
            break
        }
        if !matched {
            summary.NoMatch++
        }
    }
    return summary, nil
}
```
