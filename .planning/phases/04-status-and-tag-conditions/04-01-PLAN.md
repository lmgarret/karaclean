```go
if c.Archived != nil {
    if b.Archived != *c.Archived {
        return false
    }

if c.Favourited != nil {
    if b.Favourited != *c.Favourited {
        return false
    }

if c.HasTag != nil {
    found := false
    for _, tag := range b.Tags {
        if tag == *c.HasTag {
            found = true
            break
        }
    }
    if !found {
        return false
    }
}

if c.LacksTag != nil {
    for _, tag := range b.Tags {
        if tag == *c.LacksTag {
            return false
        }
    }
}
```

Key implementation details:
- Boolean checks: dereference pointer `*c.Archived` and compare to `b.Archived` (Go compiler enforces type safety)
- Tag matching: case-sensitive exact match with `==` (per user decision -- no strings.EqualFold)
- HasTag: linear scan, return false if not found; nil/empty Tags slice naturally produces zero iterations so `found` stays false -> correct
- LacksTag: linear scan, return false immediately if found; nil/empty Tags slice produces zero iterations -> never returns false -> correct (bookmark lacks the tag)
- No `if b.Tags == nil` guards -- Go range over nil slice is safe and produces zero iterations
