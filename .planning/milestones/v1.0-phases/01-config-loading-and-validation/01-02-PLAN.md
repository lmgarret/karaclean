   ```go
   package config

   import (
       "fmt"
       "strings"
   )
   ```

   Define `ValidationError` struct:
   ```go
   type ValidationError struct {
       Field   string
       Message string
   }

   func (e ValidationError) Error() string {
       return fmt.Sprintf("%s: %s", e.Field, e.Message)
   }
   ```

   Define `ValidationErrors` struct (implements error interface):
   ```go
   type ValidationErrors struct {
       Errors []ValidationError
   }

   func (e *ValidationErrors) Error() string {
       var b strings.Builder
       b.WriteString("config validation failed:\n")
       for _, err := range e.Errors {
           fmt.Fprintf(&b, "  - %s\n", err.Error())
       }
       return b.String()
   }
   ```

   Define valid enum values as package-level variables:
   ```go
   var validActions = []string{"archive", "delete"}
   var validSources = []string{"rss", "web", "api", "mobile", "extension", "cli", "import"}
   ```

   Implement `func (c *Config) Validate() []ValidationError`:
   - Initialize `var errs []ValidationError`
   - Check `len(c.Rules) == 0`: append error `{Field: "rules", Message: "at least one rule required"}`
   - For each rule at index i:
     - prefix := `fmt.Sprintf("rules[%d]", i)`
     - Check `rule.Action == ""`: append `{Field: prefix, Message: "missing required field \"action\""}`
     - Check `rule.Action != ""` and not in validActions: append `{Field: prefix + ".action", Message: fmt.Sprintf("invalid value %q (must be archive or delete)", rule.Action)}`
     - Check `rule.Conditions == nil`: append `{Field: prefix, Message: "missing required field \"conditions\""}`
     - If `rule.Conditions != nil`, check all fields are nil (OlderThan, Source, Archived, Favourited, HasTag, LacksTag all nil): append `{Field: prefix + ".conditions", Message: "at least one condition required"}`
     - If `rule.Conditions != nil && rule.Conditions.Source != nil`, check `*rule.Conditions.Source` is in validSources: if not, append `{Field: prefix + ".conditions.source", Message: fmt.Sprintf("invalid value %q (must be %s)", *rule.Conditions.Source, strings.Join(validSources, ", "))}`
     - If `rule.Conditions != nil && rule.Conditions.OlderThan != nil && *rule.Conditions.OlderThan <= 0`: append `{Field: prefix + ".conditions.olderThan", Message: "must be a positive number of days"}`
   - Return errs
